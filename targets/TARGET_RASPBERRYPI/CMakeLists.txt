# Generated headers ----------------------------------------------------------------------------------------------------

# The Pico SDK relies on a couple of generated files in order to work.
# We generate those files here.

if(MCU_RP2 IN_LIST MBED_TARGET_LABELS)
	# Version header -- needs version information from CMake.
	# Helpfully, that info lives in its own file, so we can just include it.
	include(pico-sdk/pico_sdk_version.cmake)
	set(PICO_GENERATED_HEADERS_DIR ${CMAKE_CURRENT_BINARY_DIR}/pico-sdk-generated)
	file(MAKE_DIRECTORY ${PICO_GENERATED_HEADERS_DIR}/pico)
	configure_file(pico-sdk/src/common/pico_base_headers/include/pico/version.h.in ${PICO_GENERATED_HEADERS_DIR}/pico/version.h)

	if(MBED_TARGET STREQUAL "RASPBERRY_PI_PICO")
		set(MBED_PICO_CONFIG_HEADER_FILE boards/pico.h)
	elseif(MBED_TARGET STREQUAL "RASPBERRY_PI_PICO_W")
		set(MBED_PICO_CONFIG_HEADER_FILE boards/pico_w.h)
	elseif(MBED_TARGET STREQUAL "SFE_THING_PLUS_RP2040")
		set(MBED_PICO_CONFIG_HEADER_FILE boards/sparkfun_thingplus.h)
	elseif(MBED_TARGET STREQUAL "RASPBERRY_PI_PICO_2")
		set(MBED_PICO_CONFIG_HEADER_FILE boards/pico2.h)
	elseif(MBED_TARGET STREQUAL "RASPBERRY_PI_PICO_2_W")
		set(MBED_PICO_CONFIG_HEADER_FILE boards/pico2_w.h)
	elseif(MBED_TARGET STREQUAL "OLIMEX_RP2350_PICO2_XL")
		set(MBED_PICO_CONFIG_HEADER_FILE boards/olimex_rp2350_xl.h)
	elseif(NOT DEFINED MBED_PICO_CONFIG_HEADER_FILE)
		message(FATAL_ERROR "Unknown RP2xxx-family target ${MBED_TARGET}, please edit TARGET_RASPBERRYPI/CMakeLists.txt or set the MBED_PICO_CONFIG_HEADER_FILE cmake variable to contain the correct Pico SDK board header file for this board (e.g. \"boards/pico.h\")")
	endif()

	# Autogen config header.
	# The following code copied from generate_config_header.cmake
	macro(add_header_content_from_var VAR)
		set(header_content "${header_content}\n\n// based on ${VAR}:\n")
		foreach(var IN LISTS ${VAR})
			set(header_content "${header_content}\n#include \"${var}\"")
		endforeach()
	endmacro()
	set(header_content "// AUTOGENERATED FROM MBED_PICO_CONFIG_HEADER_FILE and then PICO_<PLATFORM>_CONFIG_HEADER_FILES\n// DO NOT EDIT!\n")
	add_header_content_from_var(MBED_PICO_CONFIG_HEADER_FILE)
	file(GENERATE
		OUTPUT ${PICO_GENERATED_HEADERS_DIR}/pico/config_autogen.h
		CONTENT "${header_content}"
	)
endif()

target_compile_definitions(mbed-mcu-rp2
	INTERFACE
		# Enables renaming interrupt vectors to their CMSIS names
		LIB_CMSIS_CORE
		PICO_CMSIS_RENAME_EXCEPTIONS

		# Don't automatically register init stuff as global constructors. We call these manually in mbed_sdk_init()
		PICO_RUNTIME_SKIP_INIT_BOOTROM_RESET
		PICO_RUNTIME_SKIP_INIT_PER_CORE_BOOTROM_RESET
		PICO_RUNTIME_SKIP_INIT_EARLY_RESETS
		PICO_RUNTIME_SKIP_INIT_USB_POWER_DOWN
		PICO_RUNTIME_SKIP_INIT_PER_CORE_ENABLE_COPROCESSORS
		PICO_RUNTIME_SKIP_INIT_CLOCKS
		PICO_RUNTIME_SKIP_INIT_POST_CLOCK_RESETS
		PICO_RUNTIME_SKIP_INIT_RP2040_GPIO_IE_DISABLE
		PICO_RUNTIME_SKIP_INIT_SPIN_LOCKS_RESET
		PICO_RUNTIME_SKIP_INIT_INSTALL_RAM_VECTOR_TABLE

		# Don't use the shared interrupt vector mechanism, which dynamically builds functions in memory to
		# call multiple IRQ handlers. This does NOT play nice with Mbed's default configuration of
		# disabling RAM execution via the MPU.
		PICO_DISABLE_SHARED_IRQ_HANDLERS
)

target_link_libraries(mbed-mcu-rp2 INTERFACE mbed-cmsis-cortex-m)

# add a link option to wrap the given function name; i.e. -Wl:wrap=FUNCNAME for gcc
function(pico_wrap_function TARGET FUNCNAME)
    target_link_options(${TARGET} INTERFACE "LINKER:--wrap=${FUNCNAME}")
endfunction()
